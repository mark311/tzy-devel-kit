* Charts

** Bar Chart

#+begin_src gnuplot :var data=demo-data-table :exports results :file bar-chart.png

set title "Bar Chart Demo"

set xlabel "Categories"
set xrange [-0.5:2.5]
set xtics 1

set ylabel "Values"
set yrange [0.0:15.0]
set ytics 1


set boxwidth 1
set style data histograms
set boxwidth 0.6
set style fill solid

plot data using 3:xtic(2) title 'col-1' \
   , data using 4:xtic(2) title 'col-2' \
   , data using 5:xtic(2) title 'col-3' \

#+end_src
